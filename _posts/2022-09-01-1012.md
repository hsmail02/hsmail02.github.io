## python
```python
import sys
sys.setrecursionlimit(20000)


T = int(input())
for t in range(T):
    M, N, K = [int(x) for x in input().split()]
    map_ = [[0 for j in range(N)] for i in range(M)]


    def erase(i, j):
        if map_[i][j]:
            map_[i][j] = 0
            if i > 0:
                erase(i - 1, j)
            if j > 0:
                erase(i, j - 1)
            if i < M - 1:
                erase(i + 1, j)
            if j < N - 1:
                erase(i, j + 1)


    cnt = 0
    for k in range(K):
        i, j = [int(x) for x in input().split()]
        map_[i][j] = 1
    for i in range(M):
        for j in range(N):
            if map_[i][j]:
                cnt += 1
                erase(i, j)
    print(cnt)
```
