## python
```python
T = int(input())
for _ in range(T):
    x1, y1, r1, x2, y2, r2 = [int(x) for x in input().split()]
    dis_sq = (x1 - x2) ** 2 + (y1 - y2) ** 2
    out_sq = (r1 + r2) ** 2
    in_sq = (r1 - r2) ** 2
    if not dis_sq and not in_sq:
        print(-1)
    elif dis_sq in [out_sq, in_sq]:
        print(1)
    elif in_sq < dis_sq < out_sq:
        print(2)
    else:
        print(0)
```
